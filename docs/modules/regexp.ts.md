import { right } from 'fp-ts/lib/Either'

const input1 = /\w+/
const input2 = new RegExp('\\w+')
assert.deepStrictEqual(regexp.decode(input1), right(input1))
assert.deepStrictEqual(regexp.decode(input2), right(input2))
